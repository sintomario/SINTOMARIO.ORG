# 🏔️ Gramática de Liderazgo en Estructura de Nodos - SINTOMARIO 100K
## Integración de los 5 Niveles de Maxwell en el Corpus de Alta Autoridad

**Versión:** 1.0  
**Fecha:** 25 de Marzo, 2026  
**Objetivo:** Implementar la **gramática de liderazgo** basada en los 5 Niveles de John Maxwell en la estructura de los 100k nodos, asegurando que cada nodo guíe al usuario en su escalada de autoridad personal.

---

## 🎯 **FILOSOFÍA DE LA GRAMÁTICA DE LIDERAZGO**

> **"Cada nodo no solo informa; transforma al usuario de receptor pasivo a líder de su propia sanación"**

### **Principios de la Gramática Maxwell**

1. **Posición (Nivel 1):** El nodo proyecta autoridad técnica que genera confianza inicial
2. **Permiso (Nivel 2):** El nodo crea relación emocional que permite exploración segura
3. **Producción (Nivel 3):** El nodo ofrece resultados tangibles que validan el método
4. **Desarrollo (Nivel 4):** El nodo empodera para autonomía y crecimiento personal
5. **Pináculo (Nivel 5):** El nodo inspira respeto y convierte al usuario en referencia


### **2. Sistema de Clasificación Automática por Nivel**

```python
class MaxwellLevelClassifier:
    def __init__(self, node_content):
        self.content = node_content
        self.level = None
        self.confidence = 0
    
    def classify(self):
        """Clasifica nodo según indicadores de gramática Maxwell"""
        indicators = {
            'level_1': self.check_level_1_indicators(),
            'level_2': self.check_level_2_indicators(),
            'level_3': self.check_level_3_indicators(),
            'level_4': self.check_level_4_indicators(),
            'level_5': self.check_level_5_indicators()
        }
        
        # Determinar nivel dominante
        max_level = max(indicators, key=indicators.get)
        self.level = int(max_level.split('_')[1])
        self.confidence = indicators[max_level]
        
        return {
            'level': self.level,
            'confidence': self.confidence,
            'indicators': indicators
        }
    
    def check_level_1_indicators(self):
        """Busca indicadores de Nivel 1: Posición"""
        score = 0
        
        # Presencia de citas científicas
        if len(self.content.get('citations', [])) >= 2:
            score += 30
        
        # Lenguaje técnico médico
        technical_terms = ['neurobiología', 'evidencia', 'estudios', 'correlación']
        if any(term in self.content.get('text', '') for term in technical_terms):
            score += 25
        
        # Estructura objetiva
        if self.content.get('tone') == 'technical':
            score += 25
        
        # Ausencia de narrativa personal
        if not self.content.get('personal_story'):
            score += 20
        
        return score
    
    def check_level_2_indicators(self):
        """Busca indicadores de Nivel 2: Permiso"""
        score = 0
        
        # Validación emocional explícita
        validation_phrases = ['entiendo', 'es válido', 'reconozco', 'te escucho']
        text = self.content.get('text', '').lower()
        matches = sum(1 for phrase in validation_phrases if phrase in text)
        score += min(matches * 10, 30)
        
        # Pronombres subjetivos
        subjective_pronouns = ['tú', 'tu', 'lo que sientes', 'tu experiencia']
        matches = sum(1 for pronoun in subjective_pronouns if pronoun in text)
        score += min(matches * 5, 25)
        
        # Historia personal o testimonio
        if self.content.get('personal_story') or self.content.get('testimonial'):
            score += 25
        
        # Tono empático
        if self.content.get('tone') == 'empathetic':
            score += 20
        
        return score
    
    def check_level_3_indicators(self):
        """Busca indicadores de Nivel 3: Producción"""
        score = 0
        
        # Ejercicio o protocolo incluido
        if self.content.get('exercise') or self.content.get('protocol'):
            score += 35
        
        # Métricas de resultado
        if self.content.get('metrics') or self.content.get('timeframe'):
            score += 25
        
        # Lenguaje de acción
        action_verbs = ['practica', 'aplica', 'implementa', 'observa']
        text = self.content.get('text', '').lower()
        matches = sum(1 for verb in action_verbs if verb in text)
        score += min(matches * 8, 25)
        
        # Testimonios de resultado
        if self.content.get('success_stories'):
            score += 15
        
        return score
    
    def check_level_4_indicators(self):
        """Busca indicadores de Nivel 4: Desarrollo"""
        score = 0
        
        # Framework educativo
        if self.content.get('framework') or self.content.get('system'):
            score += 30
        
        # Herramientas de auto-evaluación
        if self.content.get('self_assessment') or self.content.get('checklist'):
            score += 25
        
        # Lenguaje de empoderamiento educativo
        empowerment_terms = ['aprenderás', 'desarrollarás', 'maestrearás', 'autonomía']
        text = self.content.get('text', '').lower()
        matches = sum(1 for term in empowerment_terms if term in text)
        score += min(matches * 8, 25)
        
        # Sistema de seguimiento
        if self.content.get('tracking_system'):
            score += 20
        
        return score
    
    def check_level_5_indicators(self):
        """Busca indicadores de Nivel 5: Pináculo"""
        score = 0
        
        # Filosofía o visión presente
        if self.content.get('philosophy') or self.content.get('vision'):
            score += 30
        
        # Componente de legado
        legacy_terms = ['legado', 'trascendencia', 'contribución', 'maestría']
        text = self.content.get('text', '').lower()
        matches = sum(1 for term in legacy_terms if term in text)
        score += min(matches * 8, 30)
        
        # Invitación a liderazgo o enseñanza
        if self.content.get('leadership_invitation') or self.content.get('teaching_opportunity'):
            score += 25
        
        # Comunidad de practicantes avanzados
        if self.content.get('advanced_community'):
            score += 15
        
        return score
```

### **3. Motor de Transición Maxwell**

```javascript
class MaxwellTransitionEngine {
  constructor() {
    this.userLevel = 1;
    this.progressionPath = [];
    this.transitions = {
      1: { to: 2, threshold: 'emotional_connection' },
      2: { to: 3, threshold: 'result_experienced' },
      3: { to: 4, threshold: 'education_completed' },
      4: { to: 5, threshold: 'autonomy_achieved' }
    };
  }
  
  trackProgression(userAction) {
    this.progressionPath.push({
      action: userAction,
      timestamp: Date.now(),
      level: this.userLevel
    });
    
    this.evaluateTransition();
  }
  
  evaluateTransition() {
    const currentTransition = this.transitions[this.userLevel];
    if (!currentTransition) return;
    
    const threshold = currentTransition.threshold;
    const achieved = this.checkThreshold(threshold);
    
    if (achieved) {
      this.suggestTransition(currentTransition.to);
    }
  }
  
  checkThreshold(threshold) {
    // Lógica de verificación de umbral
    const checks = {
      'emotional_connection': () => this.hasEmotionalEngagement(),
      'result_experienced': () => this.hasCompletedExercise(),
      'education_completed': () => this.hasLearnedFramework(),
      'autonomy_achieved': () => this.hasDemonstratedAutonomy()
    };
    
    return checks[threshold] ? checks[threshold]() : false;
  }
  
  suggestTransition(nextLevel) {
    const transitionData = {
      from: this.userLevel,
      to: nextLevel,
      message: this.getTransitionMessage(nextLevel),
      cta: this.getTransitionCTA(nextLevel),
      url: `/nivel-${nextLevel}/${this.getCurrentTopic()}`
    };
    
    this.renderTransitionSuggestion(transitionData);
  }
  
  getTransitionMessage(level) {
    const messages = {
      2: "Has sido visto. Ahora descubre cómo transformar esta comprensión...",
      3: "Entiendes tu experiencia. Es hora de aplicar protocolos prácticos...",
      4: "Has obtenido resultados. Desarrolla tu autonomía somática...",
      5: "Eres maestro de tu cuerpo. Comparte tu luz con otros..."
    };
    
    return messages[level] || "Continúa tu viaje de maestría...";
  }
  
  getTransitionCTA(level) {
    const ctas = {
      2: "Explora la dimensión emocional →",
      3: "Aplica el protocolo práctico →",
      4: "Domina tu autonomía →",
      5: "Reclama tu liderazgo →"
    };
    
    return ctas[level] || "Continuar →";
  }
}
```
