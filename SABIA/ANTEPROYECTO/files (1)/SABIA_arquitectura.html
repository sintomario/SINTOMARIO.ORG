Schema de datos — el ID que construye todo

// content/readings/espalda-baja-carga.json
{
  "id":        "espalda-baja-carga",     → URL · slug · filename
  "zona":      "espalda-baja",           del SVG táctil
  "contexto":  "carga",                  del selector de frase
  "emocion":   "no soy suficiente",      — visible al usuario —
  "lectura":   "Tu cuerpo está...",      120-180 palabras narrativa
  "micro":     "Pon la mano aquí...",    acción concreta < 5 min

  // internos — invisibles al usuario —
  "herida":    "rechazo",               Bourbeau → match terapia
  "conflicto": "desvalorizacion",       Hamer → tono de lectura
  "tcm":       "agua",                  elemento TCM
  "fase":      "cronica",               fase biológica Hamer
  "ref_autor": "Bourbeau",
  "ref_libro": "Tu cuerpo dice amame",  link afiliado Amazon
  "nodos":     ["hombro-carga", "cadera-carga", "espalda-baja-soledad"],
  "keywords":  ["dolor espalda emocional", "biodescodificacion espalda baja"],
  "premium":   true
}

// URL generada automáticamente desde el id:
sabia.info/espalda-baja-carga         lectura gratuita
sabia.info/api/og?id=espalda-baja-carga OG image dinámica
